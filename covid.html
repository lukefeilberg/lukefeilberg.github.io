Mapping COVID-19 Cases

I saw a map somewhere online showing the number of confirmed cases of COVID-19 per million people in each state, but I wanted to see the same map evolve with time so I made that here! The data comes from the Johns Hopkins CSSE Github repository joined with census population data.

def get_cases_per_1000000(row, date):
    return 1000000 * row[date] / row['pop']

per = usa[['geometry']]

dates = list(usa.columns)[5:]

for i in range(len(dates)):
    per[dates[i]] = usa.apply(
        lambda row: get_cases_per_1000000(row, dates[i]), 
        axis=1)

Plotting most recent infections per state population¶

In [13]:
vmin = 0
vmax = np.max(per.drop(columns='geometry').values)

def map(col):
    
    fig, ax = plt.subplots(1,figsize=(20,12));
    
    per.plot(column=col, cmap='OrRd', linewidth=0.8, ax=ax, edgecolor='0.8', vmin=vmin ,vmax=vmax)
    
    ax.axis('off');
    ax.set_title('Cases per 1,000,000 People', fontdict={'fontsize': '26', 'fontweight' : '3'})
    
    sm = plt.cm.ScalarMappable(cmap='OrRd',norm=plt.Normalize(vmin=vmin ,vmax=vmax))
    sm._A = []
    cbar = fig.colorbar(sm,cax = fig.add_axes([0.9, .25, 0.03, 0.5])) # left,bottom,width,height
    cbar.ax.tick_params(labelsize=16)
    
    fig.text(.475, .225, col, fontdict={'fontsize': '24', 'fontweight' : '3'}) #ha='center', )
    
    return fig,ax
